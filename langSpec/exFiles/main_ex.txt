func main int ()
{
    fib int = fibonacci(7);
    fac int = factorial(5);

    print:out(fib);
    print:out(fac);

    0;
}

func fibonacci int (n int)
{
    {(0) -> 1;}
    fibonacci(n-1) + fibonacci(n-2);
}

func factorial int (n int)
{
    {(n <= 1) -> 1;}
    n * factorial(n-1);
}
