// Structures are one of the main data types in Solace. Similar to objects,
// structures dictate a simplistic grouping of data values under a user defined
// structure type.
// A cake can be described as a simple collection of different elements.
struct Cake
{
    icing string,
    batter string,
    candles bool,
}

func bakeACake Cake ()
{
    // return a defined chocolate cake structure.
    chocolateCake Cake = {"Vanilla", "Chocolate", true};
}
