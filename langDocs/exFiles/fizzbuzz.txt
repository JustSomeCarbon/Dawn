module:FizzBuzz

// pattern blocks are used to define possible behaviour for
// simple pattern matching. Pattern are defined left of ->
// while designated functionality are defined to the right.
func fizzbuzz string (n int)
{
    {(0) -> "";}
    {(n % 3 == 0) -> fizzbuzz(n-1) + "\nfizz"}
    {(n % 5 == 0) -> fizzbuzz(n-1) + "\nbuzz"}
    {(n%5 == 0) && (n%3 == 0) -> fizzbuzz(n-1) + "\nfizzbuzz"}
    fizzbuzz(n-1) + "\nn";
}
